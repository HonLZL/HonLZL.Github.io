1 手写 strcpy

把 src 所指向的字符串复制到 dest。要注意的是如果目标数组 dest 不够大，而源字符串的长度又太长，可能会造成缓冲溢出的情况。

该函数返回一个指向最终的目标字符串 dest 的指针。

char* strcpy(char* dest, const char* src) {
    assert( (dest != nullptr) && (src != nullptr));
    char* p = dest;
    while ((*p++ = *src++) != '\0');
    // 写开
    // while (*src != '\0') {
    //     *p = *src;
    //     p++; src++;
    // }
    // *p = '\0';
    return dest;
}
int main() {
    char src[] = "Hello, World!";
    char dest[5] = {0};  // 目标数组 dest 不够大,会把数据写入到我们无法控制的地址中去
    strcpy(dest, src);
    printf("复制后的字符串: %s\n", dest);
    return 0;
}

所以有了 strncpy, 设置最大复制长度 n

char* strcpy(char* dest, const char* src, size_t n) {
    assert( (dest != nullptr) && (src != nullptr));
    char* p = dest;
    while (n-- && (*p++ = *src++) != '\0'); // 不能保证dest 以 '\0' 结尾
    while (n-- > 0) {
        *p++ = '\0';
    }
    return dest;
}
int main() {
    char src[] = "Hello, World!";
    char dest[5] = {0}; 
    strcpy(dest, src, sizeof(dest));
    printf("复制后的字符串: %s\n", dest);
    return 0;
}

2 手写 memcpy

void* memcpy(void* dest, const void* src, size_t n) {
    // 将源和目标指针分别强制转换为 char* 类型，因为char是C语言中最小的可寻址单位，这样可以确保按字节逐个复制数据。
    unsigned char* d = (unsigned char*) dest;
    unsigned char* s = (unsigned char*) src;
    
    for (size_t i = 0; i < n; i++) {
        d[i] = s[i];
    }
    return dest;
}

地址重叠问题是指在复制过程中，源内存区域和目标内存区域有重叠部分，可能导致复制的结果不正确。

地址堆叠 例如 

char str[] = "Hello, world!";
// 将字符串向右移动两个位置，实现地址重叠
memcpy(str + 2, str, strlen(str) + 1); // 输出为 HeHeHeHeHeHeHeHe，已经复制的 He 被继续复制到后面的内存

3 手写 memove

有重叠时，从后向前复制 

void* memmove(void* dest, const void* src, size_t n) {
    unsigned char* d = (unsigned char*)dest;
    const unsigned char* s = (const unsigned char*)src;

    // 检查源和目标内存区域是否有重叠
    if (d > s && d < s + n) {
        // 目标内存区域与源内存区域重叠，从后向前复制
        // for (size_t i = n - 1; i >= 0; --i) 死循环
        // size_t 是无符号，当 i=0 时 --， 得到的其实是 -1 + 32768 = 32767
        // 所以也可以写成  for (size_t i = n - 1; i != -1; --i)
        for (size_t i = n; i > 0; --i) {
            d[i - 1] = s[i - 1];
        }
    } else {
        // 目标内存区域与源内存区域不重叠，从前向后复制
        for (size_t i = 0; i < n; ++i) {
            d[i] = s[i];
        }
    }
    return dest;
}

4 单例模式

将单例对象设置为类私有的静态成员，静态成员只能被声明和定义一次，后面使用 get 方法获取这个实例
可以保证在整个应用程序的生命周期中，该类只有一个实例

将构造函数也设置为私有，保证后面不会再去构造这个类了

饿汉- 线程安全

class Singleton {
private:
    static Singleton* instance;

    // 私有构造函数，防止外部直接创建对象
    Singleton() {}

public:
    // 获取单例实例的静态方法
    static Singleton* getInstance() {
        return instance;
    }

    // 示例方法
    void showMessage() {
        std::cout << "Hello, Singleton!" << std::endl;
    }
};

// 在类外初始化静态成员变量
Singleton* Singleton::instance = new Singleton();

懒汉- 线程不安全，获取单例的时候才去新建它，使用双重检查锁来确保线程安全

class Singleton {
private:
  static Singleton* instance;
  static std::mutex mtx;
  Singleton(){
    cout << " contructor " << endl;
  }
public:
  static Singleton* getInstance() {
    if (!instance) {
      std::lock_guard<std::mutex> lock(mtx);
      // 双重检查，保证只有需要创建的时候，才锁住
      if (!instance) {  
        instance = new Singleton();
      }
    }
    return instance;
  }
};

// 静态成员变量需要在类外初始化
Singleton* Singleton::instance = nullptr;
std::mutex Singleton::mtx;

int main() {
  Singleton* instance =  Singleton::getInstance();
  return 0;
}

5 深拷贝

深浅拷贝同常针对于指针，其他类型的复制都是深拷贝

指针的浅拷贝是指两个元素的地址是一样的，没有重新分配内存并复制指针所指向的数据。
深拷贝是重新申请内存了，地址也不一样

class Obj {
private:
    int a[10];
    char* b;

public:
    // 构造函数
    Obj() {
        b = new char[100]; // 分配内存
        std::strcpy(b, "Default string");
    }

    // 复制构造函数
    Obj(const Obj& other) {
        // 深拷贝数组成员 a
        std::memcpy(a, other.a, sizeof(a));

        // 深拷贝字符串成员 b
        b = new char[std::strlen(other.b) + 1]; // 分配内存
        std::strcpy(b, other.b);
    }

    // 析构函数
    ~Obj() {
        delete[] b; // 释放内存
    }
}

// 为什么使用ref类内的 拷贝 去释放资源呢

class Counter { 
public: 
    int uses = 1;    // use_count
    int weaks = 1;   // weak_count + (use_count != 0) 
};

template <typename T>
class weak_ptr;

template <typename T>
class shared_ptr {
public:
    // 默认构造函数，创建一个空的shared_ptr
    shared_ptr() : ptr_(nullptr), counter_(new Counter()) {}

    // 从原始指针构造shared_ptr
    explicit shared_ptr(T* ptr) : ptr_(ptr), counter_(new Counter()) {}

    // 拷贝构造函数
    shared_ptr(const shared_ptr& other)
        : ptr_(other.ptr_), counter_(other.counter_)
    {
        ++counter_->uses;
    }

    // 移动构造函数
    shared_ptr(shared_ptr&& other) noexcept
        : ptr_(other.ptr_), counter_(other.counter_)
    {
        other.ptr_ = nullptr;
        other.counter_ = nullptr;
    }

    // 从 weak_ptr 构造 shared_ptr
    shared_ptr(const weak_ptr<T>& other)
    {
        if (!other.expired()) {
            ptr_ = other.ptr_;
            counter_ = other.counter_;
            ++counter_->uses;
        }
    }

    // 赋值操作符
    shared_ptr& operator=(const shared_ptr& other)
    {
        reset();
        ptr_ = other.ptr_;
        counter_ = other.counter_;
        ++counter_->uses;
        return *this;
    }

    // 移动赋值操作符
    shared_ptr& operator=(shared_ptr&& other) noexcept
    {
        reset();
        ptr_ = other.ptr_;
        counter_ = other.counter_;
        other.ptr_ = nullptr;
        other.counter_ = nullptr;
        return *this;
    }

    // 析构函数
    ~shared_ptr()
    {
        reset();
    }

    // 重置shared_ptr
    void reset()
    {
        if (counter_) {
            if (--counter_->uses == 0) {
                if (ptr_) {
                    delete ptr_;
                    ptr_ = nullptr;
                }
                if (counter_->weaks == 0) {
                    delete counter_;
                    counter_ = nullptr;
                }
            }
        }
    }
    
    int use_count() const { return counter_ ? counter_->uses : 0; }

    // 获取原始指针
    T* get() const { return ptr_; }

    // 解引用操作符
    T& operator*() const { return *ptr_; }

    // 操作符->
    T* operator->() const { return ptr_; }

    // 检查shared_ptr是否为空
    bool empty() const { return ptr_ == nullptr; }

private:
    T* ptr_ = nullptr;
    Counter* counter_ = nullptr;
    friend class weak_ptr<T>;
};

template <typename T>
class weak_ptr {
public:
    // 默认构造函数，创建一个空的weak_ptr
    weak_ptr() : ptr_(nullptr), counter_(nullptr) {}

    // 从shared_ptr构造weak_ptr
    weak_ptr(const shared_ptr<T>& other)
        : ptr_(other.ptr_), counter_(other.counter_)
    {
        if (counter_)
            ++counter_->weaks;
    }

    // 拷贝构造函数
    weak_ptr(const weak_ptr& other)
        : ptr_(other.ptr_), counter_(other.counter_)
    {
        if (counter_)
            ++counter_->weaks;
    }

    // 移动构造函数
    weak_ptr(weak_ptr&& other) noexcept
        : ptr_(other.ptr_), counter_(other.counter_)
    {
        other.ptr_ = nullptr;
        other.counter_ = nullptr;
    }

    // 赋值操作符
    weak_ptr& operator=(const weak_ptr& other)
    {
        reset();
        ptr_ = other.ptr_;
        counter_ = other.counter_;
        if (counter_)
            ++counter_->weaks;
        return *this;
    }

    // 移动赋值操作符
    weak_ptr& operator=(weak_ptr&& other) noexcept
    {
        reset();
        ptr_ = other.ptr_;
        counter_ = other.counter_;
        other.ptr_ = nullptr;
        other.counter_ = nullptr;
        return *this;
    }

    // 从shared_ptr赋值
    weak_ptr& operator=(const shared_ptr<T>& other)
    {
        reset();
        ptr_ = other.ptr_;
        counter_ = other.counter_;
        if (counter_)
            ++counter_->weaks;
        return *this;
    }

    // 析构函数
    ~weak_ptr()
    {
        reset();
    }

    // 重置weak_ptr
    void reset()
    {
        if (counter_) {
            // 断开与对象的关联
            if (counter_->uses == 0 && --counter_->weaks == 0) {  
                delete counter_;
                counter_ = nullptr;

            }
        }
    }

    // 获取指向对象的shared_ptr，如果对象已销毁，则返回空shared_ptr
    shared_ptr<T> lock() const
    {
        if (!counter_ || counter_->uses == 0)
            return shared_ptr<T>();
        // 这样创建的shared_ptr将增加对象的强引用计数
        return shared_ptr<T>(*this);
    }

    // 检查weak_ptr是否有效（即是否可以成功lock）
    bool expired() const { 
        return !counter_ || counter_->uses == 0; 
    }

private:
    T* ptr_ = nullptr;
    Counter* counter_ = nullptr;
    friend class shared_ptr<T>;
};

struct TestObject {
    int value;
    TestObject(int v) : value(v) {}
    ~TestObject() { std::cout << "Object with value " << value << " destroyed.\n"; }
};

int main() {
    // 创建一个TestObject实例，并用shared_ptr管理
    shared_ptr<TestObject> sptr(new TestObject(42));

    // 输出强引用计数
    std::cout << "Initial use_count: " << sptr.use_count() << "\n";

    
    shared_ptr<TestObject> sptr2 = sptr;
    std::cout << "Initial use_count: " << sptr.use_count() << "\n";



    // 从sptr构造一个weak_ptr
    weak_ptr<TestObject> wptr(sptr);

    // 输出强引用计数
    std::cout << "After creating weak_ptr, use_count: " << sptr.use_count() << "\n";

    // 检查wptr是否过期
    if (wptr.expired()) {
        std::cout << "Weak pointer is already expired.\n";
    } else {
        std::cout << "Weak pointer is not expired.\n";
    }
    return 0;
}
